def load_and_clean_data(path):
    df = pd.read_csv(path)
    df.columns = [c.strip().title() for c in df.columns]
    df.dropna(inplace=True)
    df['Date'] = pd.to_datetime(df['Date'])
    df.sort_values('Date', inplace=True)
    df.reset_index(drop=True, inplace=True)
    return df

def prepare_data(df, features=['Open','High','Low','Close','Adj Close','Volume'], window_size=60):
    scaler = MinMaxScaler(feature_range=(0,1))
    scaled_data = scaler.fit_transform(df[features])

    X, y = [], []
    for i in range(window_size, len(scaled_data)):
        X.append(scaled_data[i-window_size:i, :])  # input sequence
        y.append(scaled_data[i, :])  # output sequence (multi-feature)
    X, y = np.array(X), np.array(y)
    return X, y, scaler

In[5]:
def build_lstm(input_shape, output_size=6):
    model = Sequential([
        LSTM(100, return_sequences=True, input_shape=input_shape),
        Dropout(0.2),
        LSTM(100, return_sequences=False),
        Dropout(0.2),
        Dense(50, activation='relu'),
        Dense(output_size)
    ])
    model.compile(optimizer='adam', loss='mse', metrics=['mae'])
    return model

In[6]:
def train_model(model, X_train, y_train):
    early_stop = EarlyStopping(monitor='val_loss', patience=10, restore_best_weights=True)
    checkpoint = ModelCheckpoint(os.path.join(MODEL_DIR, "best_ohlcv_model.keras"), save_best_only=True)
    history = model.fit(
        X_train, y_train,
        epochs=50,
        batch_size=32,
        validation_split=0.1,
        callbacks=[early_stop, checkpoint],
        verbose=1
    )
    return model, history

def predict_and_update(df, model, scaler, features=['Open','High','Low','Close','Adj Close','Volume'], window_size=60):
    last_window = df[features].values[-window_size:]
    scaled_window = scaler.transform(last_window)
    X_input = np.reshape(scaled_window, (1, window_size, len(features)))

    predicted_scaled = model.predict(X_input)
    predicted = scaler.inverse_transform(predicted_scaled)[0]

    next_date = df['Date'].iloc[-1] + timedelta(days=1)
    new_row = {features[i]: predicted[i] for i in range(len(features))}
    new_row['Date'] = next_date

    df = pd.concat([df, pd.DataFrame([new_row])], ignore_index=True)
    df.to_csv(CSV_PATH, index=False)

    print(f"\n Next Day Prediction Added ({next_date.strftime('%Y-%m-%d')}):")
    for i, f in enumerate(features):
        print(f"{f}: {predicted[i]:,.2f}")
    print("\n File updated successfully at:", CSV_PATH)
    return df

def main():
    print("Loading & Cleaning Data...")
    df = load_and_clean_data(CSV_PATH)

    print(" Preparing Data for LSTM...")
    features = ['Open','High','Low','Close','Adj Close','Volume']
    X, y, scaler = prepare_data(df, features=features)

    print(" Building Model...")
    model = build_lstm((X.shape[1], X.shape[2]), output_size=len(features))

    print(" Training Model...")
    model, _ = train_model(model, X, y)

    # Save model & scaler
    model.save(os.path.join(MODEL_DIR, "final_ohlcv_model.keras"))
    model.save(os.path.join(MODEL_DIR, "final_ohlcv_model.h5"))
    joblib.dump(scaler, os.path.join(MODEL_DIR, "ohlcv_scaler.save"))
    print(" Model & Scaler Saved!")

    print(" Making Next Prediction & Updating CSV...")
    updated_df = predict_and_update(df, model, scaler, features=features)

    print("\n Process Complete! File updated successfully.")
    print("Saved at:", CSV_PATH)

if __name__ == "__main__":
    main()

import numpy as np
import pandas as pd
import joblib
import os
import matplotlib.pyplot as plt
from sklearn.metrics import mean_absolute_error, mean_squared_error, r2_score
from tensorflow.keras.models import load_model

# ---------------------------
# CONFIG
# ---------------------------
CSV_PATH = r"C:\Fyp Data\new data\BTC-USD (2014-2024).csv"
MODEL_PATH = r"C:\Fyp Data\saved_model_btc_ohlcv\final_ohlcv_model.keras"
SCALER_PATH = r"C:\Fyp Data\saved_model_btc_ohlcv\ohlcv_scaler.save"
WINDOW_SIZE = 60
FEATURES = ['Open','High','Low','Close','Adj Close','Volume']

# ---------------------------
# Load Data & Scaler
# ---------------------------
def load_data():
    df = pd.read_csv(CSV_PATH)
    df['Date'] = pd.to_datetime(df['Date'])
    df.sort_values('Date', inplace=True)
    df.reset_index(drop=True, inplace=True)
    return df

def prepare_test_data(df, scaler, window_size=WINDOW_SIZE, test_size=0.2):
    scaled_data = scaler.fit_transform(df[FEATURES].values)
    split_index = int(len(scaled_data) * (1 - test_size))

    test_data = scaled_data[split_index - window_size:]
    X_test, y_test = [], []
    for i in range(window_size, len(test_data)):
        X_test.append(test_data[i-window_size:i, :])  # input sequence
        y_test.append(test_data[i, :])  # output sequence (all features)

    X_test, y_test = np.array(X_test), np.array(y_test)
    return X_test, y_test, split_index

# ---------------------------
# Evaluation Metrics
# ---------------------------
def evaluate_model(model, X_test, y_test, scaler):
    y_pred_scaled = model.predict(X_test)
    y_pred = scaler.inverse_transform(y_pred_scaled)
    y_true = scaler.inverse_transform(y_test)

    mse = mean_squared_error(y_true, y_pred)
    mae = mean_absolute_error(y_true, y_pred)
    rmse = np.sqrt(mse)
    r2 = r2_score(y_true, y_pred)

    print("\n Model Evaluation Results (BTC OHLCV):")
    print(f" Mean Absolute Error (MAE): {mae:.2f}")
    print(f" Root Mean Squared Error (RMSE): {rmse:.2f}")
    print(f" R Score: {r2:.4f}")

    # Visualization for each feature
    for i, feature in enumerate(FEATURES):
        plt.figure(figsize=(10,4))
        plt.plot(y_true[:, i], color='blue', label=f'Actual {feature}')
        plt.plot(y_pred[:, i], color='red', label=f'Predicted {feature}')
        plt.title(f'BTC {feature} Prediction vs Actual')
        plt.xlabel('Days (Test Period)')
        plt.ylabel(feature)
        plt.legend()
        plt.grid(True)
        plt.tight_layout()
        plt.show()

# ---------------------------
# Main
# ---------------------------
def main():
    print(" Loading BTC dataset...")
    df = load_data()

    print(" Loading trained BTC OHLCV model and scaler...")
    model = load_model(MODEL_PATH)
    scaler = joblib.load(SCALER_PATH)

    print(" Preparing test dataset...")
    X_test, y_test, split_index = prepare_test_data(df, scaler)

    print(" Evaluating model...")
    evaluate_model(model, X_test, y_test, scaler)

if __name__ == "__main__":
    main()
